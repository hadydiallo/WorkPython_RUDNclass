# prediction 
def prediction(X,w,b) :
    return stepfunction((np.matmul(X,w)+b)[0])

# The perceptron
def perceptronStep(X,Y,w,b, alpha=0.1) :
    for i in range(len(X)): 
        y_pred=prediction(X[i],w,b)
        if (Y[i]-y_pred==1) :
            w[0]+=X[i][0]*alpha
            w[1]+=X[i][1]*alpha
            b+=alpha
        elif (Y[i]-y_pred==-1) :
            w[0]-=X[i][0]*alpha
            w[1]-=X[i][1]*alpha
            b-=alpha
    return w,b

# Training perceptron Algorithm

def trainPerceptronAlgo(X,Y,nb_iteration,alpha=0.1) :
    x_max=max(X.T[0])
    w=np.array(np.random.rand(2,1))
    b=np.random.rand(1)[0]+x_max
    boundary_lines=[]
    for i in range(nb_iteration) :
        w,b=perceptronStep(X,Y,w,b,alpha)
        boundary_lines.append(((-w[0]/w[1]),-b/w[1]))
    return boundary_lines

def perceptronStep(X,Y,w,b, alpha=0.1) :
    for i in range(len(X)): 
        y_pred=prediction(X[i],w,b)
        if (Y[i][0]==Y[i][1]): 
            if (Y[i][0]-y_pred==1) :
                w[0]+=X[i][0]*alpha
                w[1]+=X[i][1]*alpha
                b+=alpha
            elif (Y[i][0]-y_pred==-1) :
                w[0]-=X[i][0]*alpha
                w[1]-=X[i][1]*alpha
                b-=alpha
        elif (Y[i][0]!=Y[i][1]) :
            if (Y[i][0]-y_pred==1) :
                w[0]+=X[i][0]*alpha
                w[1]+=X[i][1]*alpha
                b+=alpha

                w[0]-=X[i][0]*alpha
                w[1]-=X[i][1]*alpha
                b-=alpha
            elif (Y[i][1]-y_pred==1) :

            elif (Y[i][1]-y_pred==-1) :

    return w,b
